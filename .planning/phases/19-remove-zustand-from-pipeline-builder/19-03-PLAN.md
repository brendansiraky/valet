```typescript
const handleDrop = (e: React.DragEvent) => {

  setIsDragOver(false);

  if (traitId) {
    addTraitToNode(id, traitId);
  }
};

const handleRemoveTrait = (traitId: string) => {
  removeTraitFromNode(id, traitId);
};
```

This is the cleanest approach - AgentNode uses the same hook that PipelineTabPanel uses. Both components share the React Query cache, and the debounced save handles persistence automatically.
